77 template <int N>
78 Eigen::Matrix<double, N, 1> AddNoiseToVector(const Eigen::Matrix<double, N, 1>& vector, const double noise_stddev) {
79  Eigen::Matrix<double, N, 1> noisy_vector = vector;
80  for (int i = 0; i < vector.rows(); ++i) {
81  noisy_vector(i, 0) += Noise(noise_stddev);
82  }
83  return noisy_vector;
84 }

86 template <int TolerancePower>
87 bool MatrixEquality(const Eigen::MatrixXd& lhs, const Eigen::MatrixXd& rhs) {
88  constexpr double tolerance = std::pow(10, -1.0 * TolerancePower);
89  // Seperately check for zero matrices since isApprox fails for these
90  if (lhs.isZero(tolerance) || rhs.isZero(tolerance)) {
91  return lhs.isZero(tolerance) && rhs.isZero(tolerance);
92  }
93  return lhs.isApprox(rhs, tolerance);
94 }
