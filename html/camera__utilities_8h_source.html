118 template <typename DISTORTER>
119 int Inliers(const std::vector<Eigen::Vector2d>& image_points, const std::vector<Eigen::Vector3d>& points_3d,
120  const Eigen::Matrix3d& intrinsics, const Eigen::VectorXd& distortion,
121  const Eigen::Isometry3d& pose_estimate, const double max_inlier_threshold,
122  boost::optional<std::vector<int>&> inliers) {
123  int num_inliers = 0;
124  for (int i = 0; i < static_cast<int>(image_points.size()); ++i) {
125  const Eigen::Vector2d& image_point = image_points[i];
126  const Eigen::Vector3d& point_3d = points_3d[i];
127  const Eigen::Vector3d transformed_point_3d = pose_estimate * point_3d;
128  const Eigen::Vector2d projected_image_point =
129  Project3dPointToImageSpaceWithDistortion<DISTORTER>(transformed_point_3d, intrinsics, distortion);
130  const Eigen::Vector2d error = (image_point - projected_image_point);
131  const double error_norm = error.norm();
132  if (error_norm <= max_inlier_threshold) {
133  ++num_inliers;
134  if (inliers) inliers->emplace_back(i);
135  }
136  }
137  return num_inliers;
138 }

149 template <typename DISTORTER>
150 boost::optional<PoseWithCovarianceAndInliers> RansacPnP(const std::vector<Eigen::Vector2d>& image_points,
151  const std::vector<Eigen::Vector3d>& points_3d,
152  const Eigen::Matrix3d& intrinsics,
153  const Eigen::VectorXd& distortion,
154  const RansacPnPParams& params) {
155  if (image_points.size() < 4) {
156  LogError("RansacPnP: Too few matched points given.");
157  return boost::none;
158  }
159 
160  const DISTORTER distorter;
161  const std::vector<Eigen::Vector2d> undistorted_image_points =
162  distorter.Undistort(image_points, intrinsics, distortion);
163 
164  std::vector<cv::Point2d> undistorted_image_points_cv;
165  for (const auto& undistorted_image_point : undistorted_image_points) {
166  undistorted_image_points_cv.emplace_back(cv::Point2d(undistorted_image_point.x(), undistorted_image_point.y()));
167  }
168  std::vector<cv::Point3d> points_3d_cv;
169  for (const auto& point_3d : points_3d) {
170  points_3d_cv.emplace_back(point_3d.x(), point_3d.y(), point_3d.z());
171  }
172 
173  cv::Mat intrinsics_cv;
174  cv::eigen2cv(intrinsics, intrinsics_cv);
175  cv::Mat rodrigues_rotation_cv(3, 1, cv::DataType<double>::type, cv::Scalar(0));
176  cv::Mat translation_cv(3, 1, cv::DataType<double>::type, cv::Scalar(0));
177  int max_num_inliers = 0;
178  Eigen::Isometry3d best_pose_estimate;
179  // Use 4 values for P3P
180  // TODO(rsoussan): make this a param
181  constexpr int kNumSampledValues = 4;
182  for (int i = 0; i < params.num_iterations; ++i) {
183  const auto sampled_indices = RandomNIndices(image_points.size(), kNumSampledValues);
184  const auto sampled_undistorted_image_points = SampledValues(undistorted_image_points_cv, sampled_indices);
185  const auto sampled_points_3d = SampledValues(points_3d_cv, sampled_indices);
186  UndistortedPnP(sampled_undistorted_image_points, sampled_points_3d, intrinsics_cv, params.pnp_method,
187  rodrigues_rotation_cv, translation_cv);
188  const Eigen::Isometry3d pose_estimate = Isometry3d(rodrigues_rotation_cv, translation_cv);
189  const int num_inliers =
190  Inliers<DISTORTER>(image_points, points_3d, intrinsics, distortion, pose_estimate, params.max_inlier_threshold);
191  if (num_inliers > max_num_inliers) {
192  best_pose_estimate = pose_estimate;
193  max_num_inliers = num_inliers;
194  }
195  }
196 
197  if (max_num_inliers < params.min_num_inliers) {
198  LogError("RansacPnP: Failed to find pose with enough inliers.");
199  return boost::none;
200  }
201 
202  std::vector<int> inliers;
203  Inliers<DISTORTER>(image_points, points_3d, intrinsics, distortion, best_pose_estimate, params.max_inlier_threshold,
204  inliers);
205  // TODO(rsoussan): Get covariance for ransac pnp
206  return PoseWithCovarianceAndInliers(best_pose_estimate, Eigen::Matrix<double, 6, 6>::Identity(), inliers);
207 }

209 template <typename DISTORTER>
210 boost::optional<PoseWithCovarianceAndInliers> ReprojectionPoseEstimate(const std::vector<Eigen::Vector2d>& image_points,
211  const std::vector<Eigen::Vector3d>& points_3d,
212  const Eigen::Vector2d& focal_lengths,
213  const Eigen::Vector2d& principal_points,
214  const Eigen::VectorXd& distortion,
215  const ReprojectionPoseEstimateParams& params) {
216  if (image_points.size() < 4) {
217  LogError("ReprojectionPoseEstimate: Too few matched points given.");
218  return boost::none;
219  }
220 
221  const Eigen::Matrix3d intrinsics = optimization_common::Intrinsics(focal_lengths, principal_points);
222  // Use RansacPnP for initial estimate since using identity transform can lead to image projection issues
223  // if any points_3d z values are 0.
224  const auto ransac_pnp_estimate =
225  RansacPnP<DISTORTER>(image_points, points_3d, intrinsics, distortion, params.ransac_pnp);
226  if (!ransac_pnp_estimate) {
227  LogError("ReprojectionPoseEstimate: Failed to get ransac pnp initial estimate.");
228  return boost::none;
229  }
230 
231  const int num_inliers = ransac_pnp_estimate->inliers.size();
232  if (num_inliers < params.ransac_pnp.min_num_inliers) {
233  LogError("ReprojectionPoseEstimate: Too few inliers found. Need " << params.ransac_pnp.min_num_inliers << ", got "
234  << num_inliers << ".");
235  return boost::none;
236  }
237 
238  return ReprojectionPoseEstimateWithInitialEstimate<DISTORTER>(
239  image_points, points_3d, focal_lengths, principal_points, distortion, params, ransac_pnp_estimate->pose,
240  ransac_pnp_estimate->inliers);
241 }

243 template <typename DISTORTER>
244 boost::optional<PoseWithCovarianceAndInliers> ReprojectionPoseEstimateWithInitialEstimate(
245  const std::vector<Eigen::Vector2d>& image_points, const std::vector<Eigen::Vector3d>& points_3d,
246  const Eigen::Vector2d& focal_lengths, const Eigen::Vector2d& principal_points, const Eigen::VectorXd& distortion,
247  const ReprojectionPoseEstimateParams& params, const Eigen::Isometry3d& initial_estimate,
248  const std::vector<int>& initial_inliers) {
249  if (image_points.size() < 4) {
250  LogError("ReprojectionPoseEstimateWithInitialEstimate: Too few matched points given.");
251  return boost::none;
252  }
253 
254  const int num_initial_inliers = initial_inliers.size();
255  if (num_initial_inliers < params.ransac_pnp.min_num_inliers) {
256  LogError("ReprojectionPoseEstimateWithInitialEstimate: Too few inliers provided. Need "
257  << params.ransac_pnp.min_num_inliers << ", got " << num_initial_inliers << ".");
258  return boost::none;
259  }
260 
261  // TODO(rsoussan): Get covariance from ransac pnp
262  if (!params.optimize_estimate)
263  return PoseWithCovarianceAndInliers(initial_estimate, Eigen::Matrix<double, 6, 6>::Identity(), initial_inliers);
264 
265  ceres::Problem problem;
266  optimization_common::AddConstantParameterBlock(2, focal_lengths.data(), problem);
267  optimization_common::AddConstantParameterBlock(2, principal_points.data(), problem);
268  optimization_common::AddConstantParameterBlock(DISTORTER::kNumParams, distortion.data(), problem);
269 
270  Eigen::Matrix<double, 6, 1> pose_estimate_vector = optimization_common::VectorFromIsometry3d(initial_estimate);
271  problem.AddParameterBlock(pose_estimate_vector.data(), 6);
272 
273  for (int i = 0; i < num_initial_inliers; ++i) {
274  const int inlier_index = initial_inliers[i];
275  optimization_common::ReprojectionError<DISTORTER>::AddCostFunction(
276  image_points[inlier_index], points_3d[inlier_index], pose_estimate_vector,
277  const_cast<Eigen::Vector2d&>(focal_lengths), const_cast<Eigen::Vector2d&>(principal_points),
278  const_cast<Eigen::VectorXd&>(distortion), problem, 1, params.optimization.huber_loss);
279  }
280 
281  ceres::Solver::Summary summary;
282  ceres::Solve(params.optimization.solver_options, &problem, &summary);
283  const Eigen::Isometry3d optimized_estimate = optimization_common::Isometry3d(pose_estimate_vector);
284  if (params.optimization.verbose) {
285  LogError("ReprojectionPoseEstimateWithInitialEstimate: Initial estimate: " << std::endl
286  << initial_estimate.matrix() << std::endl
287  << "Optimized estimate: " << std::endl
288  << optimized_estimate.matrix()
289  << std::endl
290  << "Summary: " << std::endl
291  << summary.FullReport());
292  }
293  if (!summary.IsSolutionUsable()) {
294  LogError("ReprojectionPoseEstimateWithInitialEstimate: Failed to find solution.");
295  return boost::none;
296  }
297 
298  ceres::Covariance::Options options;
299  ceres::Covariance covariance(options);
300  std::vector<std::pair<const double*, const double*>> covariance_blocks;
301  covariance_blocks.push_back(std::make_pair(pose_estimate_vector.data(), pose_estimate_vector.data()));
302  if (!covariance.Compute(covariance_blocks, &problem)) {
303  LogError("ReprojectionPoseEstimateWithInitialEstimate: Failed to compute covariances.");
304  return boost::none;
305  }
306 
307  localization_common::PoseCovariance pose_covariance;
308  covariance.GetCovarianceBlock(pose_estimate_vector.data(), pose_estimate_vector.data(), pose_covariance.data());
309 
310  std::vector<int> inliers;
311  const Eigen::Matrix3d intrinsics = optimization_common::Intrinsics(focal_lengths, principal_points);
312  Inliers<DISTORTER>(image_points, points_3d, intrinsics, distortion, optimized_estimate, params.max_inlier_threshold,
313  inliers);
314  return PoseWithCovarianceAndInliers(optimized_estimate, pose_covariance, inliers);
315 }
